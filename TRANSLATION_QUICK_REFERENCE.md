## Helper Functions

```typescript
import { formatRelativeTime, formatCurrency, formatDate } from '@/translations';

// Relative time
formatRelativeTime(new Date(timestamp), language)
// → "5 menit yang lalu" or "5 minutes ago"

// Currency
formatCurrency(1500000, language)
// → "Rp 1.500.000" or "IDR 1,500,000"

// Date
formatDate(new Date(), language)
// → "24 Februari 2025" or "February 24, 2025"
```

## Usage in Components

```typescript
import { useTranslation } from '@/hooks/useTranslation';

function MyComponent() {
  const { t, language } = useTranslation();
  
  return (
    <div>
      <h1>{t.projectDetail.title}</h1>
      <button>{t.common.save}</button>
    </div>
  );
}
```
